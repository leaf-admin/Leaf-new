// Configuração de sampling baseada em ambiente
function getSamplingRate() {
    const env = process.env.NODE_ENV || 'development';
    
    switch (env) {
        case 'production':
            // Sampling 1-5% em produção
            return parseFloat(process.env.OTEL_SAMPLING_RATE || '0.01');
        case 'staging':
            return 0.10; // 10%
        case 'development':
        default:
            return 1.0; // 100%
    }
}
